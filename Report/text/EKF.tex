\begin{equation}
\mu_{t}=A_t \mu_{t-1} + B_t u_t
\end{equation}
\begin{equation}
\Sigma_{t}=A_t \Sigma_{t-1} A_t^T +R_t
\end{equation}
\begin{equation}
K_t = \Sigma_{t} C_t(C_t^T \Sigma_{t}C_t^T+Q_t)^{-1}
\end{equation}
\begin{equation}
\mu_{t}=\mu_{t}+K_t(z_t-C_t /mu_t)
\end{equation}
\begin{equation}
\Sigma_{t}=(I-K_t C_t)\Sigma_{t}
\end{equation}
\begin{equation}
return  \mu_t ,  \Sigma_t
\end{equation}
